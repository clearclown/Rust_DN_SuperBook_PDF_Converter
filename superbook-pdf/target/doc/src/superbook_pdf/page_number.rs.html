11/// Page number detection error types
12#[derive(Debug, Error)]
13pub enum PageNumberError {
14    #[error("Image not found: {0}")]
15    ImageNotFound(PathBuf),
16
17    #[error("OCR failed: {0}")]
18    OcrFailed(String),
19
20    #[error("No page numbers detected")]
21    NoPageNumbersDetected,
22
23    #[error("Inconsistent page numbers")]
24    InconsistentPageNumbers,
25
26    #[error("IO error: {0}")]
27    IoError(#[from] std::io::Error),
28}
29
30pub type Result<T> = std::result::Result<T, PageNumberError>;
31
32/// Page number detection options
33#[derive(Debug, Clone)]
34pub struct PageNumberOptions {
35    /// Search region (percentage of image height to search)
36    pub search_region_percent: f32,
37    /// OCR language
38    pub ocr_language: String,
39    /// Minimum confidence threshold
40    pub min_confidence: f32,
41    /// Detect numbers only
42    pub numbers_only: bool,
43    /// Position hint
44    pub position_hint: Option<PageNumberPosition>,
45}

53            numbers_only: true,

228    /// Detect page number from single image
229    pub fn detect_single(
230        image_path: &Path,
231        page_index: usize,
232        options: &PageNumberOptions,
233    ) -> Result<DetectedPageNumber> {
234        if !image_path.exists() {
235            return Err(PageNumberError::ImageNotFound(image_path.to_path_buf()));
236        }
237
238        let img = image::open(image_path)
239            .map_err(|_| PageNumberError::ImageNotFound(image_path.to_path_buf()))?;
240
241        let (width, height) = img.dimensions();
242
243        // Determine search region based on position hint
244        let (search_y, search_height) = match options.position_hint {
245            Some(PageNumberPosition::TopCenter | PageNumberPosition::TopOutside) => {
246                let h = (height as f32 * options.search_region_percent / 100.0) as u32;
247                (0, h)
248            }
249            _ => {
250                let h = (height as f32 * options.search_region_percent / 100.0) as u32;
251                (height.saturating_sub(h), h)
252            }
253        };
254
255        // Crop search region
256        let search_region = img.crop_imm(0, search_y, width, search_height);
257
258        // For now, use simple image analysis instead of Tesseract
259        // In a full implementation, this would call tesseract OCR
260        let (number, raw_text, confidence) =
261            Self::analyze_region_for_numbers(&search_region, options);
262
263        Ok(DetectedPageNumber {
264            page_index,
265            number: if confidence >= options.min_confidence {
266                number
267            } else {
268                None
269            },
270            position: PageNumberRect {
271                x: 0,
272                y: search_y,
273                width,
274                height: search_height,
275            },
276            confidence: confidence / 100.0,
277            raw_text,
278        })
279    }

295    /// Analyze multiple images
296    pub fn analyze_batch(
297        images: &[PathBuf],
298        options: &PageNumberOptions,
299    ) -> Result<PageNumberAnalysis> {
300        let detections: Vec<DetectedPageNumber> = images
301            .par_iter()
302            .enumerate()
303            .map(|(i, path)| Self::detect_single(path, i, options))
304            .collect::<Result<Vec<_>>>()?;
305
306        // Analyze pattern
307        let (position_pattern, odd_offset, even_offset) = Self::analyze_pattern(&detections);
308
309        // Find missing and duplicate pages
310        let detected_numbers: Vec<i32> = detections.iter().filter_map(|d| d.number).collect();
311        let missing_pages = Self::find_missing_pages(&detected_numbers);
312        let duplicate_pages = Self::find_duplicate_pages(&detected_numbers);
313
314        let overall_confidence = if detections.is_empty() {
315            0.0
316        } else {
317            detections.iter().map(|d| d.confidence).sum::<f32>() / detections.len() as f32
318        };
319
320        Ok(PageNumberAnalysis {
321            detections,
322            position_pattern,
323            odd_page_offset_x: odd_offset,
324            even_page_offset_x: even_offset,
325            overall_confidence,
326            missing_pages,
327            duplicate_pages,
328        })
329    }

331    /// Analyze position pattern from detections
332    fn analyze_pattern(detections: &[DetectedPageNumber]) -> (PageNumberPosition, i32, i32) {
333        // Analyze X positions of detected page numbers
334        let mut odd_positions: Vec<i32> = Vec::new();
335        let mut even_positions: Vec<i32> = Vec::new();
336
337        for detection in detections {
338            if let Some(num) = detection.number {
339                let center_x = detection.position.x as i32 + detection.position.width as i32 / 2;
340                if num % 2 == 1 {
341                    odd_positions.push(center_x);
342                } else {
343                    even_positions.push(center_x);
344                }
345            }
346        }
347
348        let odd_avg = if odd_positions.is_empty() {
349            0
350        } else {
351            odd_positions.iter().sum::<i32>() / odd_positions.len() as i32
352        };
353
354        let even_avg = if even_positions.is_empty() {
355            0
356        } else {
357            even_positions.iter().sum::<i32>() / even_positions.len() as i32
358        };
359
360        // Determine pattern based on position difference
361        let position_pattern = if (odd_avg - even_avg).abs() < 50 {
362            PageNumberPosition::BottomCenter
363        } else if odd_avg > even_avg {
364            PageNumberPosition::BottomOutside
365        } else {
366            PageNumberPosition::BottomInside
367        };
368
369        (position_pattern, odd_avg, even_avg)
370    }

372    /// Find missing page numbers
373    fn find_missing_pages(numbers: &[i32]) -> Vec<usize> {
374        if numbers.is_empty() {
375            return vec![];
376        }
377
378        let min = *numbers.iter().min().unwrap();
379        let max = *numbers.iter().max().unwrap();
380        let set: HashSet<_> = numbers.iter().collect();
381
382        (min..=max)
383            .filter(|n| !set.contains(n))
384            .map(|n| (n - min) as usize)
385            .collect()
386    }
387
388    /// Find duplicate page numbers
389    fn find_duplicate_pages(numbers: &[i32]) -> Vec<i32> {
390        let mut seen = HashSet::new();
391        numbers
392            .iter()
393            .filter(|n| !seen.insert(*n))
394            .cloned()
395            .collect()
396    }

398    /// Calculate offset correction
399    pub fn calculate_offset(
400        analysis: &PageNumberAnalysis,
401        _image_width: u32,
402    ) -> Result<OffsetCorrection> {
403        let page_offsets: Vec<(usize, i32)> = analysis
404            .detections
405            .iter()
406            .enumerate()
407            .filter_map(|(i, d)| {
408                d.number.map(|num| {
409                    let offset = if num % 2 == 1 {
410                        analysis.odd_page_offset_x
411                    } else {
412                        analysis.even_page_offset_x
413                    };
414                    (i, offset)
415                })
416            })
417            .collect();
418
419        let unified_offset = if !page_offsets.is_empty() {
420            page_offsets.iter().map(|(_, o)| *o).sum::<i32>() / page_offsets.len() as i32
421        } else {
422            0
423        };
424
425        Ok(OffsetCorrection {
426            page_offsets,
427            unified_offset,
428        })
429    }

431    /// Validate page order
432    pub fn validate_order(analysis: &PageNumberAnalysis) -> Result<bool> {
433        let numbers: Vec<i32> = analysis
434            .detections
435            .iter()
436            .filter_map(|d| d.number)
437            .collect();
438
439        if numbers.len() < 2 {
440            return Ok(true);
441        }
442
443        // Check if numbers are in ascending order
444        for i in 1..numbers.len() {
445            if numbers[i] <= numbers[i - 1] {
446                return Ok(false);
447            }
448        }
449
450        Ok(true)
451    }

453    /// Parse Roman numeral to integer
454    pub fn parse_roman_numeral(text: &str) -> Option<i32> {
455        let text = text.to_lowercase().trim().to_string();
456        let roman_map = [
457            ("m", 1000),
458            ("cm", 900),
459            ("d", 500),
460            ("cd", 400),
461            ("c", 100),
462            ("xc", 90),
463            ("l", 50),
464            ("xl", 40),
465            ("x", 10),
466            ("ix", 9),
467            ("v", 5),
468            ("iv", 4),
469            ("i", 1),
470        ];
471
472        let mut result = 0;
473        let mut remaining = text.as_str();
474
475        for (numeral, value) in &roman_map {
476            while remaining.starts_with(numeral) {
477                result += value;
478                remaining = &remaining[numeral.len()..];
479            }
480        }
481
482        if remaining.is_empty() && result > 0 {
483            Some(result)
484        } else {
485            None
486        }
487    }
